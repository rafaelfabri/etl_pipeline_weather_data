class restaurante:
     restaurantes = []


     #self é um nome reservado para o método que utilizamos para dizer que cada objeto tem sua info
     
     #self é a referencia da instancia do objeto que estamos utilizando naquele momento
     
     def __init__(self, nome, cnpj):
     #atributos que queremos para classe restaurante 
     
	     self.nome = nome
	     self.cnpj = cnpj
	     #é esperado que as pessoas nao mudem a localizacao do restaurante como convencao colocamos _ (DEFINE COMO ATRIBUTO PRIVADO)
	     self._localizacao = ''
	     restaurante.restaurantes.append(nome)
	     
	     
     def __str__(self):
     	     return f'{self.nome}'
	
     #metodo para simplificar o print do objeto
     def listar_restaurantes():
     	     for restaurante_i in restaurante.restaurantes:
     	     	print(restaurante_i)
